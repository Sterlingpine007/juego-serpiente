using Newtonsoft.Json;

internal class Program
{
    private static void Main(string[] args)
    {
        string url = "https://pokeapi.co/api/v2/berry/2";

        HttpClient client = new HttpClient();

        HttpResponseMessage response = client.GetAsync(url).Result;
        string jsonResponse = response.Content.ReadAsStringAsync().Result;


        var pokeApiObject = JsonConvert.DeserializeObject<Pokemon>(jsonResponse);

        int size = pokeApiObject.size;
        string firmnessName = pokeApiObject.firmness.name;


        Console.WriteLine("Size > " + size);
        Console.WriteLine("FirmneesName > " + firmnessName);
        Console.WriteLine("Message > " + response.StatusCode);


    }
}
